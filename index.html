function copyUrl() { // 指定されたGitHub PagesのURLをセット const shareUrl = 'https://ichirohns-jpg.github.io/shiki.gomap-001/'; const text = '🚚 志木キッチンカー GO MAP\n' + '志木市のキッチンカーは今どこ？今日のご飯が見つかるリアルタイム地図。\n' + shareUrl; if (navigator.clipboard && navigator.clipboard.writeText) { navigator.clipboard.writeText(text).then(function() { alert('リンクをコピーしました'); }).catch(function() { fallbackCopy(text); }); } else { fallbackCopy(text); } } function fallbackCopy(text) { const t = document.createElement('textarea'); document.body.appendChild(t); t.value = text; t.select(); document.execCommand('copy'); document.body.removeChild(t); alert('リンクをコピーしました'); }
